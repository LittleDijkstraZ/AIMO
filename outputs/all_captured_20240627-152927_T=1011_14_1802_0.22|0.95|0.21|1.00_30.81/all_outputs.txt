    f = f.subs(y, y_sol)

    # Find the derivative of the function
    f_prime = f.diff(x)

    # Find the critical points
    critical_points = solve(f_prime, x)

    # Evaluate the function at the critical points
    values = [f.subs(x, cp) for cp in critical_points]
