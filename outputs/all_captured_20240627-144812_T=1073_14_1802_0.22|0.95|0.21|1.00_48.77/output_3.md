from sympy import symbols, solve, simplify, Abs

def area_of_polygon():
    """The points $\left(x, y\right)$ satisfying $((\vert x + y \vert - 10)^2 + ( \vert x - y \vert - 10)^2)((\vert x \vert - 8)^2 + ( \vert y \vert - 8)^2) = 0$ enclose a convex polygon. What is the area of this convex polygon?"""
    x, y = symbols('x y')

    # The given equation is satisfied when either of the two factors is zero.
    # The first factor is zero when |x + y| = 10 or |x - y| = 10.
    # The second factor is zero when |x| = 8 or |y| = 8.

    # For the first factor, we have four equations:
    # 1) x + y = 10
    # 2) x + y = -10
    # 3) x - y = 10
    # 4) x - y = -10

    # For the second factor, we have four equations:
    # 5) x = 8
    # 6) x = -8
    # 7) y = 8
    # 8) y = -8

    # Solve these equations to get the vertices of the polygon
    vertices = []
    for i in range(1, 5):
        for j in range(5, 9):
            vertices.extend(solve([eval(f'Abs(x + y) - 10'), eval(f'Abs(x - y) - 10')], (x, y)))
            vertices.extend(solve([eval(f'Abs(x) - 8'), eval(f'Abs(y) - 8')], (x, y)))

    # Remove duplicate vertices
    vertices = list(set(vertices))

    # Calculate the area of the polygon
    area = 0
    for i in range(len(vertices)):
        x1, y1 = vertices[i]
        if i == len(vertices) - 1 or vertices[i + 1] is None:
            continue
        x2, y2 = vertices[i + 1]
        area += 0.5 * abs(x1*y2 - x2*y1)

    return simplify(area)

result = area_of_polygon()
print(result)
