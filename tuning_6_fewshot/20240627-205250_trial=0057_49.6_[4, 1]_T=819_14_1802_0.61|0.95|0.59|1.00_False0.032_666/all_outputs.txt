def area_of_polygon():
    """The points $\left(x, y\right)$ satisfying $((\vert x + y \vert - 10)^2 + ( \vert x - y \vert - 10)^2)((\vert x \vert - 8)^2 + ( \vert y \vert - 8)^2) = 0$ enclose a convex polygon. What is the area of this convex polygon?"""
    x, y = symbols('x y')

    # The equation is satisfied when either of the factors is zero.
    # For the first factor to be zero, |x + y| - 10 = 0 or |x - y| - 10 = 0.
    # For the second factor to be zero, |x| - 8 = 0 or |y| - 8 = 0.
    # Solving these equations gives the vertices of the polygon.

    vertices = []
    for i in range(4):
        for j in range(4):
            if i == 0:
                eq1 = Abs(x + y) - 10
            elif i == 1:
                eq1 = Abs(x - y) - 10
            elif i == 2:
                eq1 = Abs(x) - 8
            else:
                eq1 = Abs(y) - 8

                eq2 = Abs(x - y) - 10

            vertices.append(solution)

    # The polygon is a square with side length 16.
